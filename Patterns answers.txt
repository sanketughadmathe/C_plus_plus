/*
Printing pattern
 * * * * *
 *       *
 *       *
 *       *
 * * * * *
 */

#include <iostream>
using namespace std;

int main()
{
  int i;
  int j;
  int n;
  cout << "Enter the value of n: " << endl;
  cin >> n;

  for (int i = 0; i < n; i++)
  {

    for (int j = 0; j < n; j++)
    {
      if (i == 0 || i == n - 1)
      {
        cout << "* ";
      }
      else if (j == 0 || j == n - 1)
      {
        cout << "* ";
      }
      else
      {
        cout << "  ";
      }
    }
    cout << endl;
  }
}

  for (int i = 0; i < n; i++)
  {

    for (int j = 0; j <= i; j++)
    {
      // cout << "* ";
      if (i == 0 || i == n - 1)
      {
        cout << (j + 1) << " ";

      else if (j == 0 || j == i)
      {
        cout << (j + 1) << " ";
      }
      else
      {
        cout << "  ";
      }
    }
    cout << endl;
  }
}

/*
Printing pattern
       *
      * *
     *   *
    *     *
   *       *
  * * * * * *
*/

#include <iostream>
using namespace std;

int main()
{
  int i;
  int j;
  int n;
  cout << "Enter the value of n: " << endl;
  cin >> n;

  for (int i = 0; i < n; i++)
  {

    // Spaces
    for (int space = 0; space < n - i - 1; space++)
    {
      cout << " ";
    }

    for (int j = 0; j <= i; j++)
    {
      // cout << "* ";

      if (i == 0 || i == n - 1)
      {
        cout << "* ";
      }
      else if (j == 0 || j == i)
      {
        cout << "* ";
      }
      else
      {
        cout << "  ";
      }
    }
    cout << endl;
  }
}


/*
Printing pattern
            *
          *   *
        *   *   *
      *   *   *   *
    *   *   *   *   *
  *   *   *   *   *   *
  *   *   *   *   *   *
    *   *   *   *   *
      *   *   *   *
        *   *   *
          *   *
            *
*/

#include <iostream>
using namespace std;

int main()
{
  int i;
  int j;
  int n;
  cout << "Enter the value of n: " << endl;
  cin >> n;

  for (int i = 0; i < n; i++)
  {

    // Spaces
    for (int space = 0; space < n - i - 1; space++)
    {
      cout << " ";
    }

    for (int j = 0; j <= i; j++)
    {
      cout << "* ";
    }
    cout << endl;
  }

  for (int i = 0; i < n; i++)
  {

    // Spaces
    for (int space = 0; space < i; space++)
    {
      cout << " ";
    }

    for (int j = 0; j <= n - i - 1; j++)
    {
      cout << "* ";
    }
    cout << endl;
  }
}
